    """
    The chk_broker function tests whether the activemq broker is running on
    the local host.


    """

    chk_acq = Popen(['/usr/bin/sh', '-c', 'pgrep -lfn activemq'], stdout=PIPE)
    acqpid = chk_acq.communicate()[0].decode()

    try:
        pidacq = [int(s) for s in acqpid.split() if s.isdigit()][0]
    except IndexError:
        pidacq = False

    if not pidacq:
        print('Restarting activemq broker.')
        Popen(['/usr/bin/sh', '-c', '/home/sel/bin/start_activemq'], stdout=PIPE)
        sleep(5)
    else:
        print('activemq already running.')

[docs]    def restart(self):
        """
        A method for restarting the ActiveMQ broker and AllSkyMediator.     

        The restart method calls the chk_broker function to determine
        whether the activemq message broker is running. If it is not, the
        broker is started. restart() will also restart the AllSkyMediator
        process based on the the age of the most recent data file.

        """
        chk_broker()
        try:
            pid = os.getpid()
            Popen('/bin/sh', '-c', 'kill ' + str(pid))
        except Exception as e:
            print(e)

        python = sys.executable
        os.execl(python, python, *sys.argv)

[docs]    def commander(self, msg, e, u):
        """
        Parse incoming message from AllSkyUI and call the relevant method
        to satisify the requested task.

        Parameters
        ----------
        msg : str
            the body of the message to be relayed via stomp
        e : char
            id of threading event variable
        u : float
            timeout parameter of non-blocking thread
        """

        if msg[1] == 'stop':
            self.on_stop()    
        elif msg[1] == 'start':
            self.on_start()   
        elif msg[1] == 'stop_start':
            self.on_stop_start()
        elif msg[1] == 'comcheck':
            self.on_comcheck()

        self.thread = e
        self.thread.wait(u)



[docs]    def on_stop(self):
        """
        Stop the allsky process at the request of AllSkyUI

        """
        asp = Popen(['/usr/bin/sh', '-c', 'pgrep allsky.py'], stdout=PIPE)
        self.aspid = asp.communicate()[0].decode()
        if not self.aspid:
            self.send_message('AllSky is not currently running')
            self.logmsg(f'{self.lineno()} : allsky is not currently running')
        else:
            self.send_message('Stopping allsky process ' + str(self.aspid.strip()))
            self.logmsg(f'{self.lineno()} : Stopping allsky process {self.aspid.strip()}')
            Popen(['/usr/bin/sh', '-c', 'pkill allsky.py'], stdout=PIPE)
            sleep(1)
            Popen(['/usr/bin/sh', '-c', 'pkill protoLoFits.py'], stdout=PIPE)
            sleep(1)
            asp = Popen(['/usr/bin/sh', '-c', 'pgrep allsky.py'], stdout=PIPE)
            self.aspid = asp.communicate()[0].decode()
            if not self.aspid:
                self.send_message('Allsky successfully stopped')
                self.logmsg(f'{self.lineno()} : allsky successfully stopped')  
            else:
                self.send_message('Allsky failed to stop')
                self.logmsg(f'{self.lineno()} : allsky failed to stop')

            asp = Popen(['/usr/bin/sh', '-c', 'pgrep allsky.py'], stdout=PIPE)
            self.aspid = asp.communicate()[0].decode()
            if self.aspid:
                self.send_message('AllSky successfully started with PID ' + str(self.aspid.strip()))
                self.logmsg(f'{self.lineno()} : allsky successfully started with PID {self.aspid.strip()}')

            elif not self.aspid:
                self.send_message('Allsky failed to start')

[docs]    def run(self):
        """
        Make a call to the MessageInterface to disseminate incoming
        messages or make a call to brokerMessage in the case of outgoing
        messages.

        """

        while True:
            try:
                if not self.brokerMessage:
                    self.mi.chk_allsky()
                else:
                    msg = ''
                    t = 0.03
                    u = 0.03
                    e = threading.Event()
                    t1 = threading.Thread(name='non-block',
                                          target=self.mi.message_interface(e, t),
                                          args=(e, t))
                    t1.start()

                    t2 = threading.Thread(name='non-block',
                                          target=self.mi.commander(self.brokerMessage[0:], e, u),
                                          args=(msg, e, u))
                    t2.start()
                    sleep(0.03)
                    e.set()
                    self.brokerMessage = {}
            except KeyError:
                pass

            slptm = 0.1 
            sleep(slptm)
