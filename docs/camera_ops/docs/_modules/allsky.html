[docs]    def set_connection(self):
        """
        Establish a connection to the camera if it is not already connected.

        """
        if self.get_connection() is True:
            self.logmsg(f'{self.lineno()} : The camera is connected.')
            print(f'{datetime.utcnow().strftime("%Y-%m-%d %H:%M:%S")} : The camera is connected.\n')
            self.take_image()
        else:
            cam.connect()
            self.logmsg(f'{self.lineno()} : The camera is connected: {self.get_connection()}')

            print(f'datetime.utcnow().strftime("%Y-%m-%d %H:%M:%S") : The camera is connected: '
                  f'{str(self.get_connection())}\n')
            self.take_image()



[docs]    def set_disconnect(self):
        """
        Disconnect from an active connection or log a message if there is

        """
        if self.get_connection() is False:

                  f'{str(self.get_connection())}\n')

[docs]    def sun_pos(self):
        """
        Return the current sun altitude and azimuth

        Returns
        -------
            (sunang, sunaz, sun_ang_set) : tuple

        """

        self.sunang = self.obss.sunang()[0]
        self.sunaz = self.obss.sunang()[1]
        with open(self.config_file) as f:
            cfg_data = json.load(f)
            self.sun_ang_set = cfg_data['sunang']['sun_ang_set']

        return self.sunang, self.sunaz, self.sun_ang_set



[docs]    def take_image(self):
        """
        Acquire data from the camera. take_image() calls the exposure() 
        method to determine the exposure time.

        """
        try:
            self.exptime = self.exposure()
        except TypeError:
            self.exptime = 0.1

        if self.sun_pos()[0] <= self.sun_pos()[2]:
            try:
                cam.take_data(self.exptime)
            except TypeError:
                self.wait_until()
        elif self.sun_pos()[0] > self.sun_pos()[2]:
            self.wait_until()

[docs]    def exposure(self):
        """
        exposure() uses a table in the form of a numpy array to determine
        exposure time when the sun angle is between -18 and -1.6 degrees.
        Otherwise, the exposure time is assumed to be 60.0 seconds for sun
        angles below -18 degrees.

        Returns
        -------
            exptime : float

        """
        exptbl = np.array([0.01, 0.01, 0.01, 0.01, 0.02, 0.03, 0.04, 0.04, 0.05, 0.06, 0.06, 0.07, 0.08, 0.09,
                           0.12, 0.24, 0.37, 0.45, 0.49, 0.58, 0.62, 0.67, 0.72, 0.80, 1.04, 1.61, 2.18, 2.61,
                           3.55, 4.6, 5.65, 6.45, 7.02, 7.62, 8.23, 8.86, 9.51, 10.18, 10.87, 11.58, 12.3,
                           13.04, 13.79, 14.57, 15.35, 16.15, 16.97, 17.8, 18.65, 19.51, 20.38, 21.26, 22.16,
                           23.06, 23.98, 24.91, 25.85, 26.8, 27.76, 28.73, 29.71, 30.7, 31.69, 32.69, 33.7,
                           34.72, 35.74, 36.77, 37.8, 38.84, 39.88, 40.93, 41.98, 43.03, 44.09, 45.14, 46.21,
                           47.27, 48.33, 49.4, 50.46, 51.53, 52.59, 53.65, 54.85, 55.89, 56.81, 57.6, 58.28,
                           58.85, 59.33, 59.71, 60.0])

        # Sun angle range after evening astronomical twilight
        if self.sun_pos()[0] <= -18.0 and self.sun_pos()[1] >= 180.0:
            self.exptime = 60.0
            self.logmsg(f'{self.lineno()} : Exposure time: {self.exptime:2.2f} seconds.')
            return self.exptime

        # Sun angle range before morning astronomical twilight
        if self.sun_pos()[0] <= -18.0 and self.sun_pos()[1] < 180.0:
            self.exptime = 60.0
            self.logmsg(f'{self.lineno()} : Exposure time: {self.exptime:2.2f} seconds.')
            return self.exptime

        if -18.0 < self.sun_pos()[0] <= -1.6:
            x = -1.5
            syntharr = []
            for i in range(0, -93, -1):
                syntharr.append(f'{x:2.4f}')
                x -= 0.1795

            def approximate(nparr, sunang):
                """
                Calculate the approximate sun angle then search the     
                exposure table for the actual value nearest the         
                approximation.                      

                """
                array = np.asarray(exptbl)
                idx = (np.abs(nparr - sunang)).argmin()
                try:
                    return array[idx], idx
                except IndexError:
                    return 60.0, 93

            self.nparr = np.asarray(syntharr, dtype=float)
            sunidx = approximate(self.nparr, self.sun_pos()[0])[1]

            try:
                self.exptime = exptbl[sunidx]
            except IndexError:
                self.exptime = 60.0
            self.logmsg(f'{self.lineno()} : Exposure time: {self.exptime:2.2f} seconds.')
            return self.exptime
        else:
            self.logmsg(f'{self.lineno()} : {self.sunang:2.2f} {self.sunaz:2.2f} Sun angle out of range.')
            sleep(60)
            self.wait_until()



[docs]    def wait_until(self):
        """
        wait_until() determines the current sun angle and starts or stops
        data acquisition based on the current sun angle.

        """
        while True:
            utdate = datetime.utcnow().strftime('%Y%m%d')
            try:
                last_file = sorted(glob(self.basedir + utdate + '_allsky.log'))[-1]
                last_file_mtime = getmtime(''.join(last_file))
                tdiff = time() - last_file_mtime
                if tdiff > 303:
                    self.logmsg(f'{self.lineno()} : Restarting script. tdiff {tdiff}')
                    Popen(['/usr/bin/sh', '-c', self.basedir + 'bin/asrestart &'], stdout=PIPE)
                    self.logmsg(f'{self.lineno()} : Restarted script. tdiff: {tdiff:3.2f}')
                    os._exit(0)
            except IndexError as e:
                self.logmsg(f'{self.lineno()} : Error: {e}')                               
                sleep(60)

            sundiff = abs(self.sun_pos()[0] - self.sun_pos()[2])

            if self.sun_pos()[0] <= self.sun_pos()[2]:
                if sundiff <= 0.2:
                    if self.get_connection() is False:
                        self.logmsg(f'{self.lineno()} : Connect to camera. {self.sunang:2.2f} {self.sunaz:2.2f}')
                        self.logmsg(f'{self.lineno()} : Proceed to take data')                                      
                        self.set_connection()
                        self.take_image()
                    elif self.get_connection() is True:
                        self.logmsg(f'{self.lineno()} : Camera alreay connected. {self.sunang:2.2f} {self.sunaz:2.2f}')
                        self.logmsg(f'{self.lineno()} : Proceed to take data')                                      
                        self.take_image()           
                elif sundiff > 0.2:
                    self.logmsg(f'{self.lineno()} : Alt: {self.sunang:2.2f} Az: {self.sunaz:2.2f} '
                                f'Taking data...')
                    self.take_image()
            elif self.sun_pos()[0] > self.sun_pos()[2]:
                if sundiff <= 0.2:
                    if self.get_connection() is True:
                        self.logmsg(f'{self.lineno()} : Disconnect from camera.')
                        self.set_disconnect()
                        self.logmsg(f'{self.lineno()} : Alt: {self.sunang:2.2f} Az: {self.sunaz:2.2f} '
                                    f'Sleeping for 60 seconds.')
                        sleep(60)
                elif -1.5 <= self.sun_pos()[0] <= 10.0:
                    print('sleeping for 60 seconds')
                    self.logmsg(f'{self.lineno()} : Alt: {self.sunang:2.2f} Az: {self.sunaz:2.2f} '
                                f'Sleeping for 60 seconds.')
                    sleep(60)
                elif self.sun_pos()[0] > 10.0:
                    print('sleeping for 300 seconds')
                    self.logmsg(f'{self.lineno()} : Alt: {self.sunang:2.2f} Az: {self.sunaz:2.2f} '
                                f'Sleeping for 300 seconds.')
                    sleep(300.0)
                else:
                    self.logmsg(f'{self.lineno()} : {self.sunang:2.2f} {self.sunaz:2.2f}')
                    self.logmsg(f'{self.lineno()} : There was an error in the logic.')
                    sleep(57.0)
