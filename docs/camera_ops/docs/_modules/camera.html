[docs]class IndiClient(PyIndi.BaseClient):
    """
    The IndiClient class, as it is used here, is a 
    reimplementation of the BaseClient class from the 
    pyindi_client module.

    Attributes
    ----------
    blobevent : obj
        access the Event class from the threading module

    seqno : str
        initialize the file sequence number to 1

    ccd_exposure : int
        initialize the CCD exposure time to 0

    basedir : str
        The path default base directory

    config_file: str
        The path and filename to the json configuration file

    logmsg : obj
        The instance variable to allow use of the logging class

    """
    def __init__(self):
        super(IndiClient, self).__init__()
        self.blobEvent = threading.Event()
        self.seqno = '0001'
        self.ccd_exposure = 0
        self.basedir = str(Path.home()) + '/'
        self.config_file = self.basedir + '.allsky/config.json'
        self.logmsg = log_obj.camera_logger

        with open(self.config_file) as f:
            cfg_data = json.load(f)
            camid = cfg_data['camera']['camid']

        self.ccd = camid
        device_ccd = self.getDevice(self.ccd)

        if not self.connectServer():
            self.logmsg(f'{self.lineno()} : No indiserver running on {self.getHost()} : {self.getPort()}')

        while not device_ccd:
            sleep(0.25)
            device_ccd = self.getDevice(self.ccd)

        ccd_connect = device_ccd.getSwitch("CONNECTION")

        while not ccd_connect:
            sleep(0.25)
            ccd_connect = device_ccd.getSwitch("CONNECTION")
            self.logmsg(f'{self.lineno()} : Retrying ccd_connect.')

        if not (device_ccd.isConnected()):
            ccd_connect[0].s = PyIndi.ISS_ON  # the "CONNECT" switch
            ccd_connect[1].s = PyIndi.ISS_OFF  # the "DISCONNECT" switch
            self.sendNewSwitch(ccd_connect)

        self.device_ccd = device_ccd

[docs]class Camera(IndiClient):
    """
    The Camera class inherits from IndiClient.BaseClient (above)
    and from the MessageInterface class from the protoLoFits
    module.

    Attributes
    ----------
    mi : obj
        instance of the MessageInterface class

    blobEvent : obj
         initializes to an empty object

    basedir : str
        The default base path

    logmsg : obj
        The instance logging class

    config_file : str
        The path and file name to the json configuration file

    encoded_data : obj base64 
        Format the data as a 64-bit string for transport via stomp

    switch : list
        A variable to hold the value of the switch type when a
        get of the object is requested. The getSwitch method is
        internal to PyIndi

    exptime : float
        The exposure time as a float when setting the exposure

    ccd_exposure : float
        A variable holding the value of an exposure time when a
        request for the object is issued. The 
        device_ccd.getNumber() method is internal to PyIndi.

    condition : bool
        Becomes true when an exposure has completed, false during
        an exposure

    running_start : float
        Time stamp represented as unix time

    bc : obj
        Instance of the BrokerComm class

    """

    def __init__(self):
        self.mi = MessageInterface()
        self.blobEvent = ''
        self.basedir = str(Path.home()) + '/'
        self.logmsg = log_obj.camera_logger
        self.config_file = self.basedir + '.allsky/config.json'
        self.encoded_data = ''
        self.switch = ''
        self.exptime, self.ccd_exposure = 0, 0
        self.condition = False
        self.running_start = time()
        self.bc = BrokerComm(self)

        ispid = Popen(['/bin/sh', '-c', 'pgrep indiserver'], stdout=PIPE)
        pidis = ispid.communicate()[0].decode()
        lineno = f'{inspect.currentframe().f_back.f_lineno:03d}'
        
        if not pidis:
            self.logmsg(f'{lineno} : The indiserver is not running. Will attempt to start now...')
            with open(self.config_file) as f:
                cfg_data = json.load(f)
                dev_driver = cfg_data['camera']['driver']

            try:
                os.mkfifo('/tmp/indififo')
                Popen(['/bin/sh', '-c', '/usr/bin/indiserver -f /tmp/indififo ' + str(dev_driver) + ' &'], stdout=PIPE)
                self.logmsg(f'{lineno} : Starting indi server')
            except FileExistsError:
                Popen(['/bin/sh', '-c', '/usr/bin/indiserver -f /tmp/indififo ' + str(dev_driver) + ' &'], stdout=PIPE)
                self.logmsg(f'{lineno} : Starting indi server')
            except OSError:
                Popen(['/bin/sh', '-c', '/usr/bin/indiserver ' + str(dev_driver) + ' &'], stdout=PIPE)
                self.logmsg(f'{lineno} : Starting indi server')

            IndiClient.__init__(self)
            self.setServer("localhost", 7624)
        else:
            self.logmsg(f'{lineno} : The indiserver is running with PID {pidis.strip()}')
            IndiClient.__init__(self)
            self.setServer("localhost", 7624)

[docs]    def connect(self):
        """
        connect() establishes a connection to the camera.
        
        """

        while not self.device_ccd:
            sleep(0.125)
            self.device_ccd = self.getDevice(self.ccd)
            self.logmsg(f'{self.lineno()} : not self.device_ccd')

        ccd_connect = self.device_ccd.getSwitch("CONNECTION")

        while not ccd_connect:
            sleep(0.125)
            ccd_connect = self.device_ccd.getSwitch("CONNECTION")
            self.logmsg(f'{self.lineno()} : ccd_connect not connected.')

        if not self.device_ccd.isConnected():
            self.logmsg(f'{self.lineno()} : not self.device_ccd.isConnected()')
            ccd_connect[0].s = PyIndi.ISS_ON  # the "CONNECT" switch
            ccd_connect[1].s = PyIndi.ISS_OFF  # the "DISCONNECT" switch
            self.sendNewSwitch(ccd_connect)

        if self.device_ccd.isConnected():
            frame_type = 'Light'
            self.set_frame_type(frame_type)
            self.logmsg(f'{self.lineno()} : set_frame_type({frame_type})')



[docs]    def disconnect(self):
        """
        disconnect() removes the connection to the camera.
        
        """

        while not self.device_ccd:
            sleep(0.125)
            self.device_ccd = self.getDevice(self.ccd)
            self.logmsg(f'{self.lineno()} : not self.device_ccd')

        ccd_connect = self.device_ccd.getSwitch("CONNECTION")

        while not ccd_connect:
            sleep(0.125)
            ccd_connect = self.device_ccd.getSwitch("CONNECTION")
            self.logmsg(f'{self.lineno()} : Not ccd_connect')

        if self.device_ccd.isConnected():
            ccd_connect[0].s = PyIndi.ISS_OFF  # the "CONNECT" switch
            ccd_connect[1].s = PyIndi.ISS_ON   # the "DISCONNECT" switch
            self.sendNewSwitch(ccd_connect)
            self.logmsg(f'{self.lineno()} : Sent disconnect() to camera')
            self.is_connected()



[docs]    def set_frame_type(self, newswitch):
        """
        Set the data frame type to Light, Bias, Dark or Flat. 
        Defaults to "Light".

        Parameters
        ----------
        newswitch : str
            sets the frame type using string names known by
            the PyIndi_Client

        """


        ccd_frame_type = self.device_ccd.getProperty("CCD_FRAME_TYPE")
        self.switch = ccd_frame_type.getSwitch()

        if newswitch == 'Light':
            self.switch[0].s = PyIndi.ISS_ON
            self.switch[1].s = PyIndi.ISS_OFF
            self.switch[2].s = PyIndi.ISS_OFF
            self.switch[3].s = PyIndi.ISS_OFF
        elif newswitch == 'Bias':
            self.switch[0].s = PyIndi.ISS_OFF
            self.switch[1].s = PyIndi.ISS_ON
            self.switch[2].s = PyIndi.ISS_OFF
            self.switch[3].s = PyIndi.ISS_OFF
        elif newswitch == 'Dark':
            self.switch[0].s = PyIndi.ISS_OFF
            self.switch[1].s = PyIndi.ISS_OFF
            self.switch[2].s = PyIndi.ISS_ON
            self.switch[3].s = PyIndi.ISS_OFF
        elif newswitch == 'Flat':
            self.switch[0].s = PyIndi.ISS_OFF
            self.switch[1].s = PyIndi.ISS_OFF
            self.switch[2].s = PyIndi.ISS_OFF
            self.switch[3].s = PyIndi.ISS_ON

        self.sendNewSwitch(self.switch)



[docs]    def get_frame_type(self):
        """
        Get the CCD frame type from the camera

        Attributes
        ----------
        frame_type.getSwitch() : obj
            Must be called to receive the current data frame type.
        
        Returns
        -------
        ccd_frame_type : int
            The return value will either be 0 or 1. 1 represents
            true.

        """


        ccd_frame_type = 'Light'  # Initialize the frame type to "Light"
        frame_type = self.device_ccd.getProperty("CCD_FRAME_TYPE")
        switch = frame_type.getSwitch()
         
        if switch[0].s == 1:
            ccd_frame_type = 'Light'
        elif switch[1].s == 1:
            ccd_frame_type = 'Bias'
        elif switch[2].s == 1:
            ccd_frame_type = 'Dark'
        elif switch[3].s == 1:
            ccd_frame_type = 'Flat'

        self.logmsg(f'{self.lineno()} : Current frame type: {ccd_frame_type}')
        return ccd_frame_type



[docs]    def take_data(self, exptime):
        """
        Send the exposure time when requesting an exposure from
        the camera.

        Parameters
        ----------
        exptime : float
            The exposure time value sent to the camera via the
            PyIndi Client.

        """

        self.exptime = exptime

        self.ccd_exposure = self.device_ccd.getNumber("CCD_EXPOSURE")
        while not self.ccd_exposure:
            sleep(0.125)
            self.ccd_exposure = self.device_ccd.getNumber("CCD_EXPOSURE")
            self.logmsg(f'{self.lineno()} : not self.ccd_exposure')

        try:
            self.ccd_exposure[0].value = float(self.exptime)
        except ValueError:
            self.ccd_exposure[0].value = 0.08

        self.logmsg(f'{self.lineno()} : Exposure Time: {self.exptime:2.4f} seconds.')

        # Define CCD1 as the camera being used
        self.setBLOBMode(PyIndi.B_ALSO, self.ccd, "CCD1")
        ccd_ccd1 = self.device_ccd.getBLOB("CCD1")

        while not ccd_ccd1:
            sleep(0.125)
            ccd_ccd1 = self.device_ccd.getBLOB("CCD1")
            self.logmsg(f'{self.lineno()} : Not ccd_ccd1')

        self.blobEvent = threading.Event()
        self.blobEvent.clear()
        self.sendNewNumber(self.ccd_exposure)
        self.logmsg(f'{self.lineno()} : self.blobEvent.is_set() {self.blobEvent.is_set()}')
        self.blobEvent.wait()

        for blob in ccd_ccd1:
            fitsdata = blob.getblobdata()

            while not fitsdata:
                sleep(0.125)
                fitsdata = blob.getblobdata()
                self.logmsg(f'{self.lineno()} : Size of blob: {len(fitsdata)}')
            else:
                self.logmsg(f'{self.lineno()} : Size of blob: {len(fitsdata)}')

            self.encoded_data = base64.b64encode(fitsdata)
            self.logmsg(f'{self.lineno()} : Size of encoded_data: {len(self.encoded_data)}')
            self.logmsg(f'{self.lineno()} : Name: {blob.name}  Size: {blob.size}  Bloblen: {blob.bloblen}')
            self.logmsg(f'{self.lineno()} : self.blobEvent.is_set() {self.blobEvent.is_set()}')
            self.blobEvent.clear()

            try:
                self.bc.sender(self.encoded_data, '/topic/Camera.fitsData')
                sleep(0.15)  # Limit sleep time to no less than 0.10 sec or data files will begin drop. 
                self.mi.on_data()
                self.logmsg(f'{self.lineno()} : Exposure completed.')
                self.condition = True
            except OSError:
                pass

        while self.condition is True:
            if self.exptime < 60.0:
                slptm = abs(57.0 - self.exptime)
                self.logmsg(f'{self.lineno()} : Pause time: {slptm:2.3f} seconds.')
            elif self.exptime == 60.0:
                slptm = 0
                self.logmsg(f'{self.lineno()} : Pause time: {slptm:2.3f} seconds.')
            else:
                slptm = 59.0 - self.exptime
                self.logmsg(f'{self.lineno()} : Should not be here. Pause time: {slptm:2.3f}')
      
            sleep(slptm)
            self.bc.sender('True',
                           '/topic/AllSkyUI.telemStat',
                           {'origin': 'telemStat'},
                           )

            if self.condition:
                return self.condition
